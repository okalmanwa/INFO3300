    <svg class="svg2" width="360" height="360"></svg>

    <script>
        /** First SVG: Drawing shapes and text inside the SVG container */
        let svg1 = d3.select(".svg1");
        svg1.append("circle")
            .attr("cx", 200)
            .attr("cy", 200)
            .attr("r", 10)
            .attr("fill", "lightgrey");
        
        svg1.append("text")
            .text("Carey")
            .attr("x", 200)
            .attr("y", 205)
            .attr("text-anchor", "middle")
            .attr("fill", "darkblue")
            .style("font-family", "Courier")
            .style("font-size", "30px");
        
        svg1.append("rect")
            .attr("x", 50)
            .attr("y", 50)
            .attr("width", 80)
            .attr("height", 80)
            .attr("fill", "lightcoral")
            .attr("stroke", "black")
            .attr("stroke-width", 3);
        
        svg1.append("rect")
            .attr("x", 270)
            .attr("y", 270)
            .attr("width", 100)
            .attr("height", 50)
            .attr("fill", "lightgreen")
            .attr("stroke", "darkgreen")
            .attr("stroke-width", 3);
        
        svg1.append("path")
            .attr("d", "M 300 80 L 350 100 L 330 160 L 270 160 L 250 100 Z")
            .attr("fill", "lightblue")
            .attr("stroke", "blue")
            .attr("stroke-width", 2);
        
        /** Second SVG: Plotting scatter data with gridlines */
        const data = [
            {"x": 1.0, "y": 9.0},
            {"x": 1.5, "y": 6.0},
            ...
        ];
        
        const svg2 = d3.select(".svg2");
        const width = 360;
        const height = 360;
        const padding = 20;
        
        const xScale = d3.scaleLinear()
            .domain([0, 10])
            .range([padding, width - padding]);
        
        const yScale = d3.scaleLinear()
            .domain([0, 10])
            .range([height - padding, padding]);
        
        /** Draw gridlines for both axes */
        for (let i = 0; i <= 10; i++) {
            svg2.append("line")
                .attr("x1", padding)
                .attr("x2", width - padding)
                .attr("y1", yScale(i))
                .attr("y2", yScale(i))
                .attr("stroke", "grey")
                .attr("stroke-width", 1);
        
            svg2.append("line")
                .attr("x1", xScale(i))
                .attr("x2", xScale(i))
                .attr("y1", padding)
                .attr("y2", height - padding)
                .attr("stroke", "grey")
                .attr("stroke-width", 1);
        }
        
        /** Plot data points on the scatter plot */
        data.forEach(d => {
            svg2.append("circle")
                .attr("cx", xScale(d.x))
                .attr("cy", yScale(d.y))
                .attr("r", 10)
                .attr("fill", "indigo");
        });
    </script>
